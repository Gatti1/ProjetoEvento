CREATE TABLE `boa_bd`.`evento` (
  `ID_EVENTO` INT NOT NULL,
  `TIPO_EVENTO` INT NULL,
  `DESC_EVENTO` VARCHAR(255) NULL,
  `DATA` DATE NULL,
  `ATIVO` BIT(1) NULL,
  `LOCAL` VARCHAR(250) NULL,
  `HORA` TIME NULL,
  `IMAGEM` BLOB NULL,
  PRIMARY KEY (`ID_EVENTO`),
  CONSTRAINT `ID_USUARIO`
    FOREIGN KEY (`ID_EVENTO`)
    REFERENCES `boa_bd`.`usuario` (`ID_USUARIO`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
